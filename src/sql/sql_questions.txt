show me all of the issues in a particular module that are:
    assigned to me
    that i am tracking
    
    select distinct
        scarab_issue_attribute_value.value, scarab_issue_attribute_value.ATTRIBUTE_ID
    from 
        scarab_issue_attribute_value, scarab_issue, scarab_module
    where
        (scarab_issue.module_id=2 and scarab_issue_attribute_value.issue_id=scarab_issue.issue_id)
    and
        (scarab_issue_attribute_value.attribute_id=2 or scarab_issue_attribute_value.attribute_id=10)
    and
        scarab_issue_attribute_value.visitor_id=1
    and
        scarab_issue_attribute_value.issue_id=1
